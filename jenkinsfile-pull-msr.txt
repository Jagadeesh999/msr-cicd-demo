pipeline {
    agent any
	stages {
	stage('Initialize'){
            steps {
                sh 'chmod -R 777 ${WORKSPACE}'
            }
        }
	stage('Login to container.softwareag.com'){
		environment {
                	SAGContainer = credentials('SAGContainer')
            	}
		steps{
		sh 'docker login -u $SAGContainer_USR -p $SAGContainer_PSW sagcr.azurecr.io'
	    }
	}
	stage('Pull base MSR image'){
            steps {
                sh "docker pull sagcr.azurecr.io/webmethods-microservicesruntime:10.15"
		echo "Downloaded latest image"
            }
        }
	}
}
